19 // Class that provides access to a LevelDB. Note that this class is frequently
20 // instantiated on the stack and then destroyed again, so instantiation has to
21 // be very cheap. Unfortunately that means, a CTxDB instance is actually just a
22 // wrapper around some global state.
23 //
24 // A LevelDB is a key/value store that is optimized for fast usage on hard
25 // disks. It prefers long read/writes to seeks and is based on a series of
26 // sorted key/value mapping files that are stacked on top of each other, with
27 // newer files overriding older files. A background thread compacts them
28 // together when too many files stack up.
29 //
30 // Learn more: http://code.google.com/p/leveldb/
31 class CTxDB
32 {
33 public:
34  CTxDB(const char* pszMode="r+");
35  ~CTxDB() {
36  // Note that this is not the same as Close() because it deletes only
37  // data scoped to this TxDB object.
38  delete activeBatch;
39  }

47  // A batch stores up writes and deletes for atomic application. When this
48  // field is non-NULL, writes/deletes go there instead of directly to disk.

60  template<typename K, typename T>
61  bool Read(const K& key, T& value)
62  {
63  CDataStream ssKey(SER_DISK, CLIENT_VERSION);
64  ssKey.reserve(1000);
65  ssKey << key;
66  std::string strValue;
67 
68  bool readFromDb = true;
69  if (activeBatch) {
70  // First we must search for it in the currently pending set of
71  // changes to the db. If not found in the batch, go on to read disk.
72  bool deleted = false;
73  readFromDb = ScanBatch(ssKey, &strValue, &deleted) == false;
74  if (deleted) {
75  return false;
76  }
77  }
78  if (readFromDb) {
79  leveldb::Status status = pdb->Get(leveldb::ReadOptions(),
80  ssKey.str(), &strValue);
81  if (!status.ok()) {
82  if (status.IsNotFound())
83  return false;
84  // Some unexpected error.
85  printf("LevelDB read failure: %s\n", status.ToString().c_str());
86  return false;
87  }
88  }
89  // Unserialize value
90  try {
91  CDataStream ssValue(strValue.data(), strValue.data() + strValue.size(),
92  SER_DISK, CLIENT_VERSION);
93  ssValue >> value;
94  }
95  catch (std::exception &e) {
96  return false;
97  }
98  return true;
99  }

101  template<typename K, typename T>
102  bool Write(const K& key, const T& value)
103  {
104  if (fReadOnly)
105  assert(!"Write called on database in read-only mode");
106 
107  CDataStream ssKey(SER_DISK, CLIENT_VERSION);
108  ssKey.reserve(1000);
109  ssKey << key;
110  CDataStream ssValue(SER_DISK, CLIENT_VERSION);
111  ssValue.reserve(10000);
112  ssValue << value;
113 
114  if (activeBatch) {
115  activeBatch->Put(ssKey.str(), ssValue.str());
116  return true;
117  }
118  leveldb::Status status = pdb->Put(leveldb::WriteOptions(), ssKey.str(), ssValue.str());
119  if (!status.ok()) {
120  printf("LevelDB write failure: %s\n", status.ToString().c_str());
121  return false;
122  }
123  return true;
124  }

126  template<typename K>
127  bool Erase(const K& key)
128  {
129  if (!pdb)
130  return false;
131  if (fReadOnly)
132  assert(!"Erase called on database in read-only mode");
133 
134  CDataStream ssKey(SER_DISK, CLIENT_VERSION);
135  ssKey.reserve(1000);
136  ssKey << key;
137  if (activeBatch) {
138  activeBatch->Delete(ssKey.str());
139  return true;
140  }
141  leveldb::Status status = pdb->Delete(leveldb::WriteOptions(), ssKey.str());
142  return (status.ok() || status.IsNotFound());
143  }

145  template<typename K>
146  bool Exists(const K& key)
147  {
148  CDataStream ssKey(SER_DISK, CLIENT_VERSION);
149  ssKey.reserve(1000);
150  ssKey << key;
151  std::string unused;
152 
153  if (activeBatch) {
154  bool deleted;
155  if (ScanBatch(ssKey, &unused, &deleted) && !deleted) {
156  return true;
157  }
158  }
159 
160 
161  leveldb::Status status = pdb->Get(leveldb::ReadOptions(), ssKey.str(), &unused);
162  return status.IsNotFound() == false;
163  }
